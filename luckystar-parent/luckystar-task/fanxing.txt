function medalNavInit(){
    var $list=$('#medal_list'),
        len=$list.find('.medalItem').length;
    maxLen=20,
        start=0;//初始索引为0
    $list.css({width:len*34});
    var $prev=$('#medal_nav .prev a'),
        $next=$('#medal_nav .next a');
    $prev.addClass('disable');
    if(len<=maxLen) $next.addClass('disable');
    $next.click(function(){
        if($(this).is('.disable')) return;
        $list.animate({left:'-='+maxLen*34},200);
        start+=maxLen;
        $prev.removeClass('disable');
        if(len-start<=maxLen) $next.addClass('disable');
    });
    $prev.click(function(){
        if($(this).is('.disable')) return;
        $list.animate({left:'+='+maxLen*34},200);
        start-=maxLen;
        $next.removeClass('disable');
        if(start==0) $prev.addClass('disable');
    });
}

//获取座驾列表
var usingMountId=0;
var Mount_data=null;
function loadAttrMountList(){
    var $box=$('#zj_ul');
    $.getJSON('/Services.php?act=Mount.MountService&mtd=getMyAllMount&args=["'+_userId+'"]',function(json){
        if(json.status==1){
            $box.html('');
            Mount_data=json.data.list;
            if(json.data.isShowMountListBox === 1&&Mount_data.length > 0){
                var cls = json.data.isShowMountList ===1 ? "":"checked";
                var str = '<div class="mb15 hide_mount"><em class="'+cls+'" onclick="hideMount(this);"></em><p style="float:left;">隐藏座驾列表</p></div>';
                $(".mount_start_tips").show();
                $box.append(str);
            }
            if(json.data.isShowMountList === 0 && json.data.isShowMountListBox === 0){
                $box.html("<div style='margin:30px auto;text-align:center;'>TA的座驾信息未对外公开。</div>");
                return false;
            }
            $.each(Mount_data,function(){
                var isoverdue=this.expireTime==0&&this.validTime==0;//过期判断
                var isusing=this.mountId==usingMountId;//使用中判断
                var unactive=this.validTime>0;//未激活
                var date=parseInt((this.expireTime==0?this.validTime:this.expireTime)/(24*60*60));//有效期剩余天数
                var s=['<li>\
                    <div title="价值：'+this.pricePerMonth+'星币/每月" class="pic '+(isusing?'current':'')+'"><img src="http://s3.fx.kgimg.com'+this.shopImage+'" width="156" height="100" /></div>\
                    <div class="name" title="'+this.mountName+'" ><img src="http://s3.fx.kgimg.com'+this.icon+'" width="16" height="16" /> '+this.mountName+'</div>'];
                if(json.data.isShowMountListBox === 1){//自己可以操作
                    s.push('<div class="time">'+(unactive?'有效期：':'有效期剩：')+date+'天</div>');
                    s.push('<div class="btns">');
                    if(isusing){//使用中
                        s.push('<button class="input-button cancel" onclick="mount_useCancel('+this.mountId+')">取消使用</button>');
                    }else if(!isoverdue){//未过期
                        s.push('<button class="input-button" onclick="mount_use('+this.mountId+')">使用</button>');
                    }else{//过期
                        s.push('<button class="input-button del" onclick="mount_del('+this.mountId+')">删除</button>');
                    }
                    s.push('<button class="input-button renew" onclick="mount_renew(this)">续费</button></div>');
                }else{
                    s.push('<div class="time"></div>');
                }
                if(isusing){//使用中
                    s.push('<div class="in-use">正在使用</div>');
                }else{
                    //s.push('<div class="out-use"></div>');
                }
                if(unactive){//未激活
                    s.push('<div class="unactive">未使用过</div>');
                }
                if(isoverdue){//过期
                    s.push('<div class="overdue"></div>');
                }
                s.push('</li>');
                $box.append(s.join(''));
            });
        }
    });
}

$(".fx_auto_month").live("focusout",function(){
    var textNumber = $(this).text();
    var singlePrise = $(this).parent().parent().parent().find($(".single_prise")).text();
    var totlePrise = $(this).parent().parent().parent().find($(".totle_prise"));
    $(this).parent().parent().parent().find($(".months em")).removeClass("current");
    var re = /^[1-9]$|^[0-1]\d$|^2[0-4]$/;
    if(!re.test(textNumber)){
        $(this).parent().parent().find($(".other_months")).removeClass("other_months_current");
        $(this).text("");
    }else{
        $(this).parent().parent().find($(".other_months")).addClass("other_months_current");
        totlePrise.text(parseInt(singlePrise)*parseInt($(this).text()));
    }
})

function buyNumber(o,autoBuyOrSend,autoMonth,autoPrise){
    var nowObj = o;
    var textNumber = $("#"+autoMonth);
    var textPrise = $("#"+autoPrise);
    var singlePrise = $("#"+autoBuyOrSend).find($(".single_prise")).text();
    if(nowObj.innerHTML == "-"){
        if(parseInt(textNumber.text())<2 || textNumber.text() ==""){
            textNumber.text("1");
            return false;
        }
        textNumber.text(parseInt(textNumber.text())-1);
        textPrise.text((parseInt(textNumber.text()))*parseInt(singlePrise))
    }else{
        if(parseInt(textNumber.text())>23 || textNumber.text() ==""){
            textNumber.text("24");
            return false;
        }
        textNumber.text(parseInt(textNumber.text())+1);
        textPrise.text((parseInt(textNumber.text()))*parseInt(singlePrise));
    }
}
